CREATE VOLATILE TABLE USERS_BASE AS 
( SELECT USER_ID , CASE WHEN USER_SITE_ID IN (0,100,2,210) THEN 'NA' WHEN USER_SITE_ID IN (71,101,186) THEN 'FRITES' WHEN USER_SITE_ID = 15 THEN 'AU' WHEN USER_SITE_ID = 3 THEN 'UK' WHEN USER_SITE_ID = 77 THEN 'DE' ELSE 'UNKNOWN' END AS USER_SITE_CODE , CASE WHEN CRM_GH_FLAG = 'N' THEN 'T' WHEN CRM_GH_FLAG = 'Y' THEN 'C' ELSE 'U' END AS VARIANT 
    FROM p_campaign_t.DIPSTICK_USERBASE_2016 A 
    WHERE MULTI_ACCT_YN = 'N' AND PRIMARY_ACCOUNT_IND = 1 AND EMAIL_MAST_PREF_YN = 'Y' AND GBH = 'N' AND USER_CRE_DATE < CAST((CURRENT_DATE - (CURRENT_DATE - DATE '1998-12-13') MOD 7) AS TIMESTAMP(0)) ) 
WITH DATA UNIQUE PRIMARY INDEX (USER_ID) 
ON COMMIT PRESERVE ROWS;
