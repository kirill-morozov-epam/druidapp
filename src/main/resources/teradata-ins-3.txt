insert into P_yzhang6_T.seller_hub_segment -- 8 seconds 
    SELECT a.slr_id, CASE WHEN CSS_SlrType IN ( 'Occasional', 'Regular', 'Entrepreneur', 'Merchant', 'Large Merchant') THEN CSS_SlrType ELSE CASE WHEN COALESCE(fs.FA_LSTG_AUCT_START_DT, DATE '2099-01-01')>=a.SRC_LAST_MDFD_DT THEN 'FirstTimeLister' ELSE 'New' END END AS CSS_SEGMENT 
    FROM ( SEL bb.slr_id, bb.SRC_LAST_MDFD_DT ,MAX(lkp.cust_sgmntn_desc) CSS_SlrType 
        FROM seller_hub_seller bb LEFT JOIN prs_restricted_v.cust_slr_sgmntn_hist hist ON bb.slr_id =hist.slr_id AND hist.cust_sgmntn_grp_cd BETWEEN 36 AND 41 AND SRC_LAST_MDFD_DT BETWEEN hist.cust_slr_sgmntn_beg_dt AND hist.cust_slr_sgmntn_end_dt LEFT OUTER JOIN prs_restricted_v.cust_sgmntn lkp on hist.CUST_SGMNTN_CD = lkp.CUST_SGMNTN_CD AND hist.CUST_SGMNTN_GRP_CD = lkp.CUST_SGMNTN_GRP_CD GROUP BY 1,2 ) a 
    INNER JOIN DW_COFALA_SU fs ON a.slr_id= fs.user_id ;
