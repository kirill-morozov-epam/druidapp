CREATE VOLATILE TABLE A AS ( 
    SELECT P.ITEM_ID, CASE WHEN LD.LSTG_ATTR_ID = 4116 THEN 'BRAND' WHEN LD.LSTG_ATTR_ID = 4111 THEN 'MPN' WHEN LD.LSTG_ATTR_ID IN (4112, 4113, 4114) THEN 'GTIN' END PRDCT_ASPCT_TYPE, LD.VALUE_DESC ASPCT_VLU_NM 
    FROM POPULATION P 
    INNER JOIN 
    ACCESS_VIEWS.DW_ATTR_LSTG_DTL LD ON LD.ITEM_ID = P.ITEM_ID AND LD.AUCT_END_DT >= CURRENT_DATE -2 AND LD.LSTG_ATTR_ID IN (4111, 4112, 4113, 4114, 4116) 
    GROUP BY 1,2,3 ) 
WITH DATA PRIMARY INDEX (ITEM_ID) ON COMMIT PRESERVE ROWS;
