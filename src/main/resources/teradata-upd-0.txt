UPDATE a 
FROM p_gda_schedules_t.JEG_313_POLICY_04 a ,(SELECT src.POLICY_ID ,MAX(CASE WHEN cwplcy.LEVEL_ID = 1 THEN cwplcy.NAME ELSE NULL END) AS L1_POLICY_NAME ,MAX(CASE WHEN cwplcy.LEVEL_ID = 2 THEN cwplcy.NAME ELSE NULL END) AS L2_POLICY_NAME ,MAX(CASE WHEN cwplcy.LEVEL_ID = 3 THEN cwplcy.NAME ELSE NULL END) AS L3_POLICY_NAME FROM ebay_ts_v.CW_POLICY cwplcy INNER JOIN (SELECT src.SITE_ID ,src.POLICY_ID ,cwpcyl.L1 ,cwpcyl.L2 ,cwpcyl.L3 FROM p_gda_schedules_t.JEG_313_POLICY_04 src INNER JOIN ebay_ts_v.CW_POLICY_LEVEL cwpcyl ON cwpcyl.L5 = src.POLICY_ID AND src.LEVEL_ID = 4)src ON cwplcy.SITE_ID = src.SITE_ID AND (cwplcy.ID = src.L1 OR cwplcy.ID = src.L2 OR cwplcy.ID = src.L3) GROUP BY 1)b 
SET L1_POLICY_NAME = b.L1_POLICY_NAME ,L2_POLICY_NAME = b.L2_POLICY_NAME ,L3_POLICY_NAME = b.L3_POLICY_NAME 
WHERE b.POLICY_ID = a.POLICY_ID AND a.LEVEL_ID = 4;
